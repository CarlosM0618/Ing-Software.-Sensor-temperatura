Crear Bot de Telegram y obtener Token

Carlos — Configurar el canal de alertas

▼

El ESP32 necesita un Bot de Telegram para enviar mensajes. Se crea en 2 minutos desde la app.

📱 PASO A PASO EN TELEGRAM

1
Abre Telegram y busca @BotFather (el bot oficial de Telegram para crear bots)
2
Escribe /newbot y presiona enviar
3
BotFather te preguntará el nombre del bot — escribe: Monitor Temperatura UAG
4
Te pedirá el username — escribe: MonitorTempEquipo2Bot (debe terminar en Bot)
5
BotFather te dará un TOKEN — guárdalo, se ve así:
7123456789:AAFxxxxxxxxxxxxxxxxxxxxxxxxxxxxx
6
Busca tu bot por su username, ábrelo y presiona START
7
Para obtener tu Chat ID entra a este link en el navegador (reemplaza TOKEN):
https://api.telegram.org/botTOKEN/getUpdates
8
En el JSON que aparece busca "id" dentro de "chat" — ese es tu Chat ID (número)
⚠️ Guarda estos dos datos en un lugar seguro — los necesitarás en el código:
• BOT_TOKEN: el token que te dio BotFather
• CHAT_ID: el número de tu chat

Conectar el ESP32 a internet y enviar mensaje a Telegram

Roberto — Primera alerta real

⚠️ A diferencia de los otros equipos, aquí el ESP32 se conecta a una red WiFi con internet real (la del salón o su celular en hotspot). NO crea su propia red.
main.py — Conectar a WiFi + enviar mensaje Telegram
import network, urequests, time
from machine import Pin
import dht

# ── Configuración — CAMBIA ESTOS VALORES ────
WIFI_SSID     = "nombre_de_tu_wifi"
WIFI_PASS     = "contraseña_wifi"
BOT_TOKEN     = "TU_TOKEN_AQUI"
CHAT_ID       = "TU_CHAT_ID_AQUI"
UMBRAL_TEMP   = 30   # °C — alerta si supera este valor

# ── Sensor DHT11 en D4 ──────────────────────
sensor = dht.DHT11(Pin(4))

# ── Conectar a WiFi ─────────────────────────
def conectar_wifi():
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)
    wlan.connect(WIFI_SSID, WIFI_PASS)
    print("Conectando a WiFi", end="")
    for _ in range(20):
        if wlan.isconnected(): break
        print(".", end="")
        time.sleep(1)
    if wlan.isconnected():
        print(f"\n✓ Conectado: {wlan.ifconfig()[0]}")
    else:
        print("\n✗ Error de conexión")

# ── Enviar mensaje a Telegram ────────────────
def enviar_telegram(mensaje):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    data = {"chat_id": CHAT_ID, "text": mensaje, "parse_mode": "HTML"}
    try:
        r = urequests.post(url, json=data)
        print("✓ Telegram enviado")
        r.close()
    except Exception as e:
        print(f"✗ Error Telegram: {e}")

# ── Programa principal ───────────────────────
conectar_wifi()

alerta_enviada = False
lectura = 1

print("=== MONITOR ACTIVO — Umbral: {UMBRAL_TEMP}°C ===")

while True:
    try:
        sensor.measure()
        temp = sensor.temperature()
        hum  = sensor.humidity()
        print(f"[{lectura}] {temp}°C | {hum}%")

        if temp >= UMBRAL_TEMP and not alerta_enviada:
            msg = (f"🚨 <b>ALERTA DE TEMPERATURA</b>\n\n"
                   f"🌡 Temperatura: <b>{temp}°C</b>\n"
                   f"💧 Humedad: {hum}%\n"
                   f"⚠️ Supera el umbral de {UMBRAL_TEMP}°C\n"
                   f"📍 ESP32 — UAG Ingeniería de Software")
            enviar_telegram(msg)
            alerta_enviada = True

        elif temp < UMBRAL_TEMP and alerta_enviada:
            enviar_telegram(f"✅ Temperatura normalizada: {temp}°C")
            alerta_enviada = False

        lectura += 1
    except OSError:
        print("⚠ Error de sensor")

    time.sleep(10)   # Leer cada 10 segundos

💧 Humedad: 58%

main.py — Sistema completo Dashboard + Telegram
import network, socket, urequests, time, dht
from machine import Pin

# ── Configuración ────────────────────────────
WIFI_SSID   = "nombre_de_tu_wifi"
WIFI_PASS   = "contraseña_wifi"
BOT_TOKEN   = "TU_TOKEN_AQUI"
CHAT_ID     = "TU_CHAT_ID_AQUI"
UMBRAL      = 30

sensor = dht.DHT11(Pin(4))
historial = []
alerta_enviada = False

def conectar_wifi():
    w = network.WLAN(network.STA_IF)
    w.active(True); w.connect(WIFI_SSID, WIFI_PASS)
    for _ in range(20):
        if w.isconnected(): break
        time.sleep(1)
    ip = w.ifconfig()[0] if w.isconnected() else "sin conexión"
    print(f"IP: {ip}")
    return ip

def telegram(msg):
    try:
        r = urequests.post(
            f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage",
            json={"chat_id":CHAT_ID, "text":msg, "parse_mode":"HTML"})
        r.close()
    except: pass

ip = conectar_wifi()

srv = socket.socket()
srv.bind(socket.getaddrinfo('0.0.0.0', 80)[0][-1])
srv.listen(1)
print(f"Dashboard en http://{ip}")

while True:
    cl, _ = srv.accept()
    try:
        sensor.measure()
        temp = sensor.temperature()
        hum  = sensor.humidity()
    except:
        temp = hum = "--"

    historial.append(temp)
    if len(historial) > 20: historial.pop(0)

    if isinstance(temp, int):
        if temp >= UMBRAL and not alerta_enviada:
            telegram(f"🚨 <b>ALERTA</b>\n🌡 {temp}°C — supera umbral de {UMBRAL}°C\n💧 Humedad: {hum}%")
            alerta_enviada = True
        elif temp < UMBRAL and alerta_enviada:
            telegram(f"✅ Normalizado: {temp}°C")
            alerta_enviada = False
        color = "#ff4444" if temp >= UMBRAL else "#00c9a7"
        estado = "⚠ TEMP ALTA — Alerta enviada" if temp >= UMBRAL else "✓ Normal"
    else:
        color="#ffd700"; estado="⚠ Sin sensor"

    hist_str = " | ".join([f"{v}°" for v in historial[-5:]])

    html = f"""HTTP/1.1 200 OK\r\nContent-Type: text/html\r\n\r\n
<!DOCTYPE html><html><head>
<meta charset='UTF-8'><meta http-equiv='refresh' content='5'>
<meta name='viewport' content='width=device-width,initial-scale=1'>
<title>Monitor Temperatura</title>
<style>
body{{background:#0d0f14;color:#e8eaf0;font-family:monospace;
     min-height:100vh;display:flex;flex-direction:column;
     align-items:center;justify-content:center;padding:2rem}}
h1{{color:#229ED9;font-size:1.4rem;margin-bottom:0.3rem;text-align:center}}
.sub{{color:#6b7280;font-size:0.7rem;margin-bottom:1.5rem;text-align:center}}
.val{{font-size:5rem;font-weight:900;color:{color};text-shadow:0 0 30px {color}40;line-height:1}}
.unit{{color:#6b7280;font-size:1.2rem;margin-bottom:0.5rem}}
.hum{{font-size:1.2rem;color:#60a5fa;margin-bottom:0.5rem}}
.estado{{font-size:0.9rem;color:{color};font-weight:bold;margin-bottom:1.5rem}}
.card{{background:#161920;border:1px solid #2a2f3d;border-radius:10px;
       padding:1rem 1.5rem;width:100%;max-width:360px;margin-top:0.75rem}}
.card-t{{color:#6b7280;font-size:0.65rem;text-transform:uppercase;letter-spacing:0.1em;margin-bottom:0.4rem}}
.hist{{color:#94a3b8;font-size:0.8rem}}
.tg{{color:#229ED9;font-size:0.75rem;margin-top:0.5rem}}
.refresh{{color:#334155;font-size:0.65rem;margin-top:1rem}}
</style></head><body>
<h1>🌡️ Monitor de Temperatura</h1>
<div class='sub'>Equipo 2 · UAG · Alertas en Telegram</div>
<div class='val'>{temp}°</div>
<div class='unit'>Celsius</div>
<div class='hum'>💧 {hum}% humedad</div>
<div class='estado'>{estado}</div>
<div class='card'>
  <div class='card-t'>Últimas lecturas</div>
  <div class='hist'>{hist_str}</div>
  <div class='tg'>📨 Umbral Telegram: {UMBRAL}°C</div>
</div>
<div class='refresh'>↻ Auto-refresh cada 5s</div>
</body></html>"""

    cl.send(html)
    cl.close()
